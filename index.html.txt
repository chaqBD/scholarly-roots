<section id="framework">
  <div class="how-grid-bg"></div>
  <div class="framework-header reveal">
    <div class="section-label" style="justify-content:center">
      <span class="section-label-line"></span>
      <span class="section-label-text">Reading Framework</span>
      <span class="section-label-line"></span>
    </div>
    <h2 class="section-heading">The <em>13-Dimension</em> Review Framework</h2>
    <p>Every article reviewed through the same rigorous lens. Two volunteer reviewers prepare this before each session.</p>
  </div>
  <div class="framework-grid reveal">
    <div class="fw-card">
      <div class="fw-num">01</div>
      <div class="fw-dimension">Bibliographic Details</div>
      <div class="fw-question">Authors, year, journal, ABS ranking, DOI, citation count.</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">02</div>
      <div class="fw-dimension">Aims & Objectives</div>
      <div class="fw-question">What does the paper set out to do? Are the aims clearly stated and achieved?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">03</div>
      <div class="fw-dimension">Research Questions</div>
      <div class="fw-question">What questions/hypotheses are addressed? Are they novel and well-grounded?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">04</div>
      <div class="fw-dimension">Theoretical Framework</div>
      <div class="fw-question">Which theories/models are used? Is the theoretical grounding appropriate?</div>
    </div>
    <div class="fw-card highlight">
      <div class="fw-num">05</div>
      <div class="fw-dimension">Methodology</div>
      <div class="fw-question">Research design, data collection, sample, analytical technique — all justified?</div>
    </div>
    <div class="fw-card highlight">
      <div class="fw-num">06</div>
      <div class="fw-dimension">Key Findings</div>
      <div class="fw-question">Main results — what is surprising, expected, or contested?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">07</div>
      <div class="fw-dimension">Theoretical Contributions</div>
      <div class="fw-question">Does this extend, challenge, or merely replicate existing theory?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">08</div>
      <div class="fw-dimension">Practical Contributions</div>
      <div class="fw-question">What do managers and leaders gain? Is it actionable?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">09</div>
      <div class="fw-dimension">Policy Implications</div>
      <div class="fw-question">What does this suggest for governments, regulators, institutions — UK or global?</div>
    </div>
    <div class="fw-card">
      <div class="fw-num">10</div>
      <div class="fw-dimension">Strengths</div>
      <div class="fw-question">What does this paper do particularly well — rigour, originality, clarity?</div>
    </div>
    <div class="fw-card highlight">
      <div class="fw-num">11</div>
      <div class="fw-dimension">Weaknesses & Limitations</div>
      <div class="fw-question">Blind spots, methodological flaws, generalisation issues, potential bias?</div>
    </div>
    <div class="fw-card highlight">
      <div class="fw-num">12</div>
      <div class="fw-dimension">Research Gaps</div>
      <div class="fw-question">What future research is called for? What PhD-worthy gaps do YOU spot?</div>
    </div>
    <div class="fw-card" style="grid-column: span 2; background: rgba(184,134,11,.1);">
      <div class="fw-num">13</div>
      <div class="fw-dimension">Community Verdict</div>
      <div class="fw-question">Score 1–10. Must-read or skip? Would you cite it? Group consensus vs. dissent — all documented.</div>
    </div>
  </div>
</section>

<script>
  // ── NAV SCROLL ──
  window.addEventListener('scroll', () => {
    document.getElementById('main-nav').classList.toggle('scrolled', window.scrollY > 40);
  });

  // ── MOBILE MENU ──
  function toggleMenu() {
    document.getElementById('mobile-menu').classList.toggle('open');
  }

  // ── SCROLL REVEAL ──
  const revealObserver = new IntersectionObserver((entries) => {
    entries.forEach((e, i) => {
      if (e.isIntersecting) {
        setTimeout(() => e.target.classList.add('visible'), i * 80);
        revealObserver.unobserve(e.target);
      }
    });
  }, { threshold: 0.1 });

  document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));

  // ── VOTE SYSTEM ──
  let selectedCandidate = null;
  let hasVoted = false;

  function selectCandidate(index) {
    if (hasVoted) return;
    selectedCandidate = index;
    document.querySelectorAll('.candidate').forEach((c, i) => {
      c.classList.toggle('selected', i === index);
    });
    const btn = document.getElementById('vote-btn');
    btn.disabled = false;
    btn.textContent = 'Cast My Vote →';
  }

  function castVote() {
    if (selectedCandidate === null || hasVoted) return;
    hasVoted = true;

    // Get current votes
    const candidates = document.querySelectorAll('.candidate');
    const voteCounts = Array.from(candidates).map(c => parseInt(c.dataset.votes) || 0);
    voteCounts[selectedCandidate] += 1;
    const total = voteCounts.reduce((a, b) => a + b, 0);

    // Update UI
    candidates.forEach((c, i) => {
      const pct = Math.round((voteCounts[i] / total) * 100);
      c.dataset.votes = voteCounts[i];
      c.querySelector('.candidate-pct').textContent = pct + '%';
      c.querySelector('.candidate-bar-fill').style.width = pct + '%';
      c.querySelector('.candidate-fill').style.width = pct + '%';
    });

    const btn = document.getElementById('vote-btn');
    btn.textContent = '✓ Vote Cast — Thank You!';
    btn.disabled = true;
    btn.style.background = 'var(--sage)';

    showToast('Your vote has been cast! Results update live.');
  }

  // ── TOAST ──
  let toastTimer;
  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => t.classList.remove('show'), 3500);
  }

  // ── JOIN FORM ──
  function handleJoin() {
    const email = document.getElementById('join-email').value.trim();
    if (!email || !email.includes('@')) {
      showToast('Please enter a valid email address.');
      return;
    }
    showToast('Welcome to Scholarly Roots! Check your inbox for next steps.');
    document.getElementById('join-email').value = '';
  }

  // ── HERO CARD ANIMATION ──
  document.querySelectorAll('.journal-card').forEach((card, i) => {
    card.style.animationDelay = (1.4 + i * 0.15) + 's';
  });
</script>
